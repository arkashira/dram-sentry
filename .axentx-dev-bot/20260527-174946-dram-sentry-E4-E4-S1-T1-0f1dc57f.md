# tests/unit/test_benchmark_calculation.py
import datetime as dt
import pytest
from analytics.benchmark import (
    calculate_90d_moving_average,
    _cache_key,
    BenchmarkCache,
)

# Helper to generate price series
def gen_prices(start: dt.date, days: int, start_price: float = 100.0):
    return [
        {"date": start - dt.timedelta(days=i), "price": start_price + i}
        for i in range(days)
    ]

def test_calculate_90d_average_happy_path():
    today = dt.date(2024, 5, 20)
    prices = gen_prices(today, 90)          # exactly 90 days
    avg = calculate_90d_moving_average(prices, today)
    # arithmetic series: avg = start_price + (days-1)/2
    assert avg == pytest.approx(100.0 + 44.5)

def test_calculate_90d_average_more_than_90_days():
    today = dt.date(2024, 5, 20)
    prices = gen_prices(today, 120)         # 120 days, only last 90 used
    avg = calculate_90d_moving_average(prices, today)
    # last 90 days start at price = 100 + 30
    assert avg == pytest.approx(130.0 + 44.5)

def test_calculate_90d_average_insufficient_data_returns_none():
    today = dt.date(2024, 5, 20)
    prices = gen_prices(today, 45)          # < 90 days
    avg = calculate_90d_moving_average(prices, today)
    assert avg is None

def test_cache_key_is_deterministic():
    key1 = _cache_key("SKU123", dt.date(2024,5,20))
    key2 = _cache_key("SKU123", dt.date(2024,5,20))
    assert key1 == key2
    assert key1 == "benchmark:SKU123:2024-05-20"

def test_benchmark_cache_store_and_ttl(monkeypatch):
    cache = BenchmarkCache(ttl_seconds=3600)
    today = dt.date(2024,5,20)
    key = _cache_key("SKU123", today)
    cache.set(key, 123.45)

    # value should be present immediately
    assert cache.get(key) == 123.45

    # fast‑forward time beyond TTL
    monkeypatch.setattr(dt, "datetime", lambda *a, **k: dt.datetime(2024,5,20,1,0,1))
    assert cache.get(key) is None

def test_calculate_90d_average_uses_cache(monkeypatch):
    """Ensures the public function respects the cache."""

    today = dt.date(2024,5,20)
    key = _cache_key("SKU123", today)
