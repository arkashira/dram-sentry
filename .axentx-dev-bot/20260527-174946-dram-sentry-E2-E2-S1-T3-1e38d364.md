REDIS_URL = os.getenv("REDIS_URL", "redis://localhost:6379/0")

METRIC_REQUESTS = Counter(

    'Total number of dashboard API requests',
    ['endpoint']

MODEL_PRECISION = Gauge(
    'model_precision',
    'Current model precision score'

    'model_recall',
    'Current model recall score'

ALERT_COUNT = Counter(
    'active_alerts_total',

    ['alert_type']

class MetricsCache:
    def __init__(self):

        self._initialize_cache()

    def _initialize_cache(self):
        """Initialize cache with default values if empty"""
        if not self.redis.exists("model_performance"):
            default_metrics = ModelMetrics(
                precision=0.95,
                recall=0.80,
                f1_score=0.87,
                last_updated=datetime.utcnow().isoformat(),
                alert_thresholds={"z_score": 3.5, "confidence": 0.9}
            )
            self.update_model_performance(default_metrics)

    def update_model_performance(self, metrics: ModelMetrics):
        """Update model performance metrics in cache"""
        self.redis.hset(
            "model_performance",
            mapping={
                "precision": metrics.precision,
                "recall": metrics.recall,
                "f1_score": metrics.f1_score,
                "last_updated": metrics.last_updated,
                "alert_thresholds": json.dumps(metrics.alert_thresholds)
            }
        )
        MODEL_PRECISION.set(metrics.precision)
        MODEL_RECALL.set(metrics.recall)

    def add_alert(self, alert: Alert):
        """Add new alert to cache"""
        alert_key = f"alert:{int(time.time())}"

                "timestamp": alert.timestamp,
                "type": alert.type,
                "value": alert.value,
                "resolved": str(alert.resolved),
                "message": alert.message
            }
        )
        self.redis.lpush("alerts:recent", alert_key)
